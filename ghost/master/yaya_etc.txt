{
	<<"

	真0 【rank!=0】：何だろう。わくわく。
	">>
}

//------------------------------------------------------------------------------
//インストール完了メッセージ
//------------------------------------------------------------------------------
OnInstallComplete
{
	reference[0] = SPLIT(reference[0],' with',2)[0] //hogehoge with hemohemo
	ref_id = ''
	if reference[0] == "shell" {
		ref_id = 'shell'
	}
	elseif reference[0] == "ghost" {
		ref_id = 'ghost'
	}
	elseif reference[0] == "balloon" {
		ref_id = 'balloon'
	}

	<<"
	＠設定：クイックセクション
	真【R0 == 'shell'】：【シェル
	真【R0 == 'ghost'】：【ゴースト
	真【R0 == 'balloon'】：【バルーン
	真【R0 == 'plugin'】：【プラグイン
	真【それ以外】：【タイプ不明
	改行なし真：、インストール完了】
	改行多め真：『%(reference[1])』
 	改行多め真【STRLEN(ref_id) > 0】：切り替えますか？
 	＠メニュー【直前】：切り替える｜OnAYATEMPLATE.InstallChange｜%(ref_id)
 	＠メニュー【直前】：切り替えない｜Menu_CANCEL
	">>
}

//------------------------------------------------------------------------------
//メールチェック開始
//------------------------------------------------------------------------------
OnBIFFBegin
{
	<<"
	真：\_q【メールチェック】\_q\w9
	真3【rank!=0】：あ、パソコンに何か出てるよ。\nメールじゃない？
	">>
}

//------------------------------------------------------------------------------
//メールチェック成功
//------------------------------------------------------------------------------
OnBIFFComplete
{
	if reference[0] == 0
	{
		//---- メールは無かった
		<<"
		真：\_q【新着メールなし】\_q
		真2【rank!=0】：あ、違った。
		">>
	}
	else
	{
		//---- メールが届いていた
		<<"
		＠設定：クイックセクション
		真：【新着メールあり】
		真【rank==0】：%(reference[0])通のメールが来ています
		＠設定：クイックセクション
		真0【rank!=0】：%(reference[0])通のメールがきてます、だって。
		">>
	}
}

//------------------------------------------------------------------------------
//メールチェック失敗
//------------------------------------------------------------------------------
OnBIFFFailure
{
	if reference[0] == "timeout"
	{
		//---- タイムアウト
		<<"
		真：\_q【メールチェック失敗：タイムアウトしました】\_q
		真9【rank!=0】：あれ、なんだこれ。
		">>
	}
	elseif reference[0] == "kick"
	{
		//---- 蹴られた
		<<"
		真：\_q【メールチェック失敗：キックされました】\_q
		真9【rank!=0】：あれ、なんだこれ。
		">>
	}
	elseif reference[0] == "defect"
	{
		//---- 設定誤り
		<<"
		真：\_q【メールチェック失敗：設定が間違っています】\_q
		真9【rank!=0】：あれ、なんだこれ。
		">>
	}
	else
	{
		//---- 原因不明
		<<"
		真：\_q【メールチェック失敗：原因不明のエラー】\_q
		真9【rank!=0】：あれ、なんだこれ。
		">>
	}
}

OnSNTPBegin
{
	<<"
	真：\_q【時計合わせ】\_q
	真4【rank!=0】：時計合わせてあげるね。\w9\w9
	">>
}

//------------------------------------------------------------------------------
//時刻取得成功、比較開始
//この関数で 2008/03/25 19:15:05 のように見やすい形に作り変えます。
//詳しくはSTRFORM関数の説明を見てね！
//わからなければこのままで。
//------------------------------------------------------------------------------

SNTPCompare_StrForm
{
	_txt = SPLIT(_argv[0],',')
	STRFORM("$04d/$02d/$02d $02d:$02d:$02d",TOINT(_txt[0]),TOINT(_txt[1]),TOINT(_txt[2]),TOINT(_txt[3]),TOINT(_txt[4]),TOINT(_txt[5]))
}

OnSNTPCompare
{
	<<"
	＠設定：クイックセクション
	真：【時計合わせ完了】
	真：修正前 %(SNTPCompare_StrForm(reference[2]))
	真：修正後 %(SNTPCompare_StrForm(reference[1]))
	＠設定：クイックセクション
	真0【rank!=0】：\n[half]終わったよ。\w9\w9
	真3【rank!=0 && TOINT(R3)==0】：\n[half]なんだ、ぴったりじゃん。
	真7【rank!=0 && TOINT(R3)!=0】：\n[half]%(reference[3])秒もずれてたね。
	">>
}
